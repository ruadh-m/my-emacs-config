** Unified Model Switching
#+BEGIN_SRC emacs-lisp
  (defvar my/ai-default-model "qwen2.5-coder:7b"
    "Default model used by all AI tools on startup.")

  (defvar my/ai-models
    '( "qwen2.5-coder:7b"
       "deepseek-coder:6.7b"
       "huihui_ai/seed-coder-abliterate:8b-instruct-q8_0"
       "granite3.3:8b"
       "llama3.1:8b" )
    "List of local model identifiers for `my/ai-model-switch'.")

  (defvar my/non-fim-models
    '("huihui_ai/seed-coder-abliterate:8b-instruct-q8_0"
      "deepseek-coder:6.7b")
    "Models that should use chat completions instead of FIM.")

  (defun my/configure-ai-tools-for-model (model)
    "Update gptel, minuet and ellama to use the local MODEL."
    (require 'gptel)
    (require 'minuet)
    (require 'llm-ollama)

    (setq gptel-backend my/gptel-ollama-backend)
    (setq gptel-model model)

    (if (member model my/non-fim-models)
        (progn
          (setq minuet-provider 'openai-compatible)
          (plist-put minuet-openai-compatible-options :end-point
                     "http://localhost:11434/v1/chat/completions")
          (plist-put minuet-openai-compatible-options :model model)
          (plist-put minuet-openai-compatible-options :api-key "TERM"))
      (setq minuet-provider 'openai-fim-compatible)
      (plist-put minuet-openai-fim-compatible-options :end-point
                 "http://localhost:11434/v1/completions")
      (plist-put minuet-openai-fim-compatible-options :model model)
      (plist-put minuet-openai-fim-compatible-options :api-key "TERM"))

    (setopt ellama-provider (make-llm-ollama :chat-model model)))

  (defun my/ai-model-switch (model)
    "Switch gptel, minuet and ellama to a local MODEL."
    (interactive
     (list (completing-read "Switch AI model to: " my/ai-models nil t)))
    (my/configure-ai-tools-for-model model)
    (message "✅ All AI tools switched to: %s" model))

  (my/configure-ai-tools-for-model my/ai-default-model)

  (global-set-key (kbd "C-c m s") #'my/ai-model-switch)
